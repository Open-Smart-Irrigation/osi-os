# SenseCAP S2120 Weather Station Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add full support for the Seeed SenseCAP S2120 8-in-1 LoRaWAN weather station across osi-os (edge) and osi-server (cloud), including data ingest, device card UI, multi-zone assignment, and advisory integration.

**Architecture:** ChirpStack decodes S2120 binary payloads using the Seeed codec in the device profile; Node-RED receives the decoded `object.data.messages[]` and maps it to `device_data`. A new `weather_station_zones` junction table enables one S2120 to serve multiple irrigation zones. S2120 temperature/humidity takes priority over KIWI sensors for VPD in the dendro advisory.

**Tech Stack:** Node-RED (flows.json JSON editing), SQLite (schema migrations embedded in flows.json), React + TypeScript + Tailwind (device card), Spring Boot / Java (osi-server backend), Flyway (PostgreSQL migrations), Mockito + JUnit 5 (server tests).

```javascript
// Process S2120 uplink — ChirpStack has already decoded via Seeed codec.
// msg.payload.object.data.messages is an array of arrays of { measurementValue, measurementId, type }
const data = msg.payload;
if (!data || !data.deviceInfo) return [null, null];

const devEui = String(data.deviceInfo.devEui || '').toUpperCase();
if (!devEui) return [null, null];

// Lookup device — must be SENSECAP_S2120
const _db = new osiDb.Database('/data/db/farming.db');
const q   = (sql) => new Promise((res,rej) => _db.all(sql, (e,r) => e?rej(e):res(r||[])));
const close = () => new Promise(res => _db.close(() => res()));

try {
  const rows = await q(`SELECT type_id FROM devices WHERE deveui='${devEui.replace(/'/g,"''")}' AND deleted_at IS NULL`);
  if (!rows[0] || rows[0].type_id !== 'SENSECAP_S2120') { await close(); return [null, null]; }
  await close();
} catch(e) { try { await close(); } catch(_){} return [null, null]; }

// Flatten messages array-of-arrays by measurementId
const measurements = {};
const messages = data.object?.data?.messages || [];
for (const group of messages) {
  if (!Array.isArray(group)) continue;
  for (const m of group) {
    if (m && m.measurementId != null) {
      measurements[String(m.measurementId)] = m.measurementValue;
    }
    // Battery is a plain key, not measurementId
    if (m && typeof m['Battery(%)'] === 'number') {
      measurements['bat_pct'] = m['Battery(%)'];
    }
  }
}

const timestamp = data.time ? new Date(data.time).toISOString() : new Date().toISOString();

const d = {
  devEui,
  timestamp,
  ambientTemperature:    measurements['4097'] ?? null,  // Air Temperature °C
  relativeHumidity:      measurements['4098'] ?? null,  // Air Humidity %
  lightLux:              measurements['4099'] ?? null,  // Light Intensity lux
  barometricPressureHpa: measurements['4101'] ?? null,  // Barometric Pressure hPa
  windDirectionDeg:      measurements['4104'] ?? null,  // Wind Direction °
  windSpeedMps:          measurements['4105'] ?? null,  // Wind Speed m/s
  uvIndex:               measurements['4190'] ?? null,  // UV Index UVI
  rainGaugeCumulativeMm: measurements['4113'] ?? null,  // Rain Gauge mm (cumulative)
  windGustMps:           measurements['4215'] ?? null,  // Peak Wind Gust m/s
  batPct:                measurements['bat_pct'] ?? null,
};

// Rain delta logic — query previous cumulative from device_data
let rainMmDelta = null;
let rainMmPerHour = null;
let rainMmToday = null;
let rainDeltaStatus = 'no_rain_sensor';

if (d.rainGaugeCumulativeMm != null) {
  const _db2 = new osiDb.Database('/data/db/farming.db');
  const q2   = (sql) => new Promise((res,rej) => _db2.all(sql, (e,r) => e?rej(e):res(r||[])));
  const close2 = () => new Promise(res => _db2.close(() => res()));
  try {
    const prev = await q2(`SELECT rain_gauge_cumulative_mm, recorded_at FROM device_data WHERE deveui='${devEui.replace(/'/g,"''")}' AND rain_gauge_cumulative_mm IS NOT NULL ORDER BY recorded_at DESC LIMIT 1`);
    if (prev[0] && prev[0].rain_gauge_cumulative_mm != null) {
      const delta = d.rainGaugeCumulativeMm - prev[0].rain_gauge_cumulative_mm;
      if (delta < 0) {
        // Device reset — treat new cumulative as first sample
        rainDeltaStatus = 'first_sample';
      } else {
        rainMmDelta = delta;
        const intervalMs = new Date(timestamp) - new Date(prev[0].recorded_at);
        if (intervalMs > 0) {
          rainMmPerHour = (delta / intervalMs) * 3600000;
        }
        rainDeltaStatus = 'ok';
      }
    } else {
      rainDeltaStatus = 'first_sample';
    }

    // Rain today — sum deltas for local calendar day
    const todayRows = await q2(`SELECT SUM(rain_mm_delta) AS total FROM device_data WHERE deveui='${devEui.replace(/'/g,"''")}' AND rain_delta_status='ok' AND date(recorded_at)=date('${timestamp}')`);
    rainMmToday = (todayRows[0]?.total ?? 0) + (rainMmDelta ?? 0);
    await close2();
  } catch(e) { try { await close2(); } catch(_){} }
  rainDeltaStatus = rainDeltaStatus || 'ok';
}

d.rainMmDelta    = rainMmDelta;
d.rainMmPerHour  = rainMmPerHour;
d.rainMmToday    = rainMmToday;
d.rainDeltaStatus = rainDeltaStatus;

msg.formattedData = d;
// Output 1: SQL insert; Output 2: rain aggregation (only if delta valid)
const rainOut = (rainDeltaStatus === 'ok' && rainMmDelta != null) ? msg : null;
return [msg, rainOut];
```

  const payload = JSON.parse(fromB64u(payB64).toString('utf8'));
  if (payload.exp && Date.now()>payload.exp) throw Object.assign(new Error('Token expired'),{statusCode:401});
  return { userId: Number(payload.userId) };
}

function fmtNum(v: unknown, decimals: number, unit: string): string {
  const n = Number(v);
  if (!Number.isFinite(n)) return '—';
  return `${n.toFixed(decimals)} ${unit}`;
}

function windDirLabel(v: unknown): string {
  const deg = Number(v);
  if (!Number.isFinite(deg)) return '—';
  const dirs = ['N','NE','E','SE','S','SW','W','NW'];
  return dirs[Math.round(deg / 45) % 8] + ' ' + Math.round(deg) + '°';
}

function fmtLux(v: unknown): string {
  const n = Number(v);
  if (!Number.isFinite(n)) return '—';
  return n >= 1000 ? `${(n / 1000).toFixed(1)}k lux` : `${Math.round(n)} lux`;
}

function lastSeenLabel(lastSeen: string | null | undefined): string {
  if (!lastSeen) return 'never';
  const diff = Math.floor((Date.now() - new Date(lastSeen).getTime()) / 60000);
  if (diff < 1) return 'just now';
  if (diff < 60) return `${diff} min ago`;
